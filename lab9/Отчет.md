### 1. Протофайлы (.proto)

**recipe_service.proto:**

```protobuf
syntax = "proto3";

package recipe.v1;

service RecipeService {
    rpc GenerateRecipe(GenerateRecipeRequest) returns (GenerateRecipeResponse);
    rpc GetRecipe(GetRecipeRequest) returns (Recipe);
    rpc RateRecipe(RateRecipeRequest) returns (RateRecipeResponse);
    rpc StreamUserRecipes(StreamUserRecipesRequest) returns (stream Recipe);
}

enum RecipeStatus {
    DRAFT = 0;
    GENERATED = 1;
    COOKING = 2;
    COMPLETED = 3;
    FAILED = 4;
    ARCHIVED = 5;
}

message Ingredient {
    string name = 1;
    optional string expiry_date = 2;
}

message CookingStep {
    int32 order = 1;
    string description = 2;
    optional int32 duration_minutes = 3;
}

message Recipe {
    string id = 1;
    string user_id = 2;
    string name = 3;
    repeated Ingredient ingredients = 4;
    repeated CookingStep steps = 5;
    RecipeStatus status = 6;
    string source = 7;
    optional int32 rating = 8;
    int64 created_at = 9;
}

message GenerateRecipeRequest {
    string user_id = 1;
    repeated string ingredient_names = 2;
}

message GenerateRecipeResponse {
    string recipe_id = 1;
    RecipeStatus status = 2;
    string message = 3;
}

message GetRecipeRequest {
    string recipe_id = 1;
}

message RateRecipeRequest {
    string recipe_id = 1;
    string user_id = 2;
    int32 rating = 3;
}

message RateRecipeResponse {
    bool success = 1;
    string message = 2;
}

message StreamUserRecipesRequest {
    string user_id = 1;
    RecipeStatus status_filter = 2;
}
```

### 2. gRPC Server

**Реализованные методы:**

| № | Метод | Тип | Описание |
|---|-------|-----|----------|
| 1 | `GenerateRecipe` | Unary | Генерация рецепта из ингредиентов |
| 2 | `GetRecipe` | Unary | Получение рецепта по ID |
| 3 | `StreamUserRecipes` | Server Streaming | Поток рецептов пользователя |

**Код gRPC сервера:**

```python
import grpc
from concurrent import futures
import recipe_pb2
import recipe_pb2_grpc


class RecipeServicer(recipe_pb2_grpc.RecipeServiceServicer):
    """Реализация gRPC сервера для Recipe Service"""
    
    def __init__(self, recipe_repository, user_repository):
        self._recipe_repo = recipe_repository
        self._user_repo = user_repository
    
    def GenerateRecipe(self, request, context):
        """Unary RPC: генерация рецепта"""
        try:
            # Проверка пользователя
            user = self._user_repo.find_by_id(request.user_id)
            if not user:
                context.set_code(grpc.StatusCode.NOT_FOUND)
                context.set_details(f"User {request.user_id} not found")
                return recipe_pb2.GenerateRecipeResponse()
            
            # Создание рецепта
            recipe = Recipe(
                user_id=request.user_id,
                name=f"Блюдо из {', '.join(request.ingredient_names[:2])}",
                ingredients=[Ingredient(name=n) for n in request.ingredient_names],
                steps=[]
            )
            
            self._recipe_repo.save(recipe)
            
            return recipe_pb2.GenerateRecipeResponse(
                recipe_id=recipe.id,
                status=recipe_pb2.GENERATED,
                message="Рецепт успешно сгенерирован"
            )
        except Exception as e:
            context.set_code(grpc.StatusCode.INTERNAL)
            context.set_details(str(e))
            return recipe_pb2.GenerateRecipeResponse()
    
    def GetRecipe(self, request, context):
        """Unary RPC: получение рецепта по ID"""
        recipe = self._recipe_repo.find_by_id(request.recipe_id)
        
        if not recipe:
            context.set_code(grpc.StatusCode.NOT_FOUND)
            context.set_details(f"Recipe {request.recipe_id} not found")
            return recipe_pb2.Recipe()
        
        return self._to_proto_recipe(recipe)
    
    def StreamUserRecipes(self, request, context):
        """Server Streaming RPC: поток рецептов пользователя"""
        recipes = self._recipe_repo.find_by_user_id(request.user_id)
        
        for recipe in recipes:
            if not context.is_active():
                break
            
            if request.status_filter and recipe.status.value != request.status_filter:
                continue
            
            yield self._to_proto_recipe(recipe)
    
    def _to_proto_recipe(self, recipe):
        """Конвертация доменного рецепта в protobuf"""
        return recipe_pb2.Recipe(
            id=recipe.id,
            user_id=recipe.user_id,
            name=recipe.name,
            ingredients=[
                recipe_pb2.Ingredient(name=i.name) for i in recipe.ingredients
            ],
            steps=[
                recipe_pb2.CookingStep(
                    order=s.order,
                    description=s.description,
                    duration_minutes=s.duration_minutes
                ) for s in recipe.steps
            ],
            status=self._map_status(recipe.status.value),
            source=recipe.source,
            rating=recipe.rating if recipe.rating else None,
            created_at=int(recipe.created_at.timestamp())
        )
    
    def _map_status(self, status_str):
        status_map = {
            "draft": recipe_pb2.DRAFT,
            "сгенерирован": recipe_pb2.GENERATED,
            "готовится": recipe_pb2.COOKING,
            "завершён": recipe_pb2.COMPLETED,
            "провален": recipe_pb2.FAILED,
            "архивирован": recipe_pb2.ARCHIVED
        }
        return status_map.get(status_str, recipe_pb2.DRAFT)


def serve():
    server = grpc.server(futures.ThreadPoolExecutor(max_workers=10))
    recipe_pb2_grpc.add_RecipeServiceServicer_to_server(
        RecipeServicer(recipe_repo, user_repo), server
    )
    server.add_insecure_port('[::]:50051')
    print("✅ gRPC сервер запущен на порту 50051")
    server.start()
    server.wait_for_termination()


if __name__ == '__main__':
    serve()
```

### 4. Server-Side Streaming

**Сценарий:**
Клиент подписывается на рецепты пользователя → сервер отправляет обновления в real-time.

**Код серверной части (Streaming):**

```python
# streaming_server.py
import grpc
from concurrent import futures
import time
import threading
import recipe_pb2
import recipe_pb2_grpc


class RecipeStreamServicer(recipe_pb2_grpc.RecipeServiceServicer):
    """Реализация gRPC сервера с поддержкой streaming"""
    
    def __init__(self, recipe_repository):
        self._recipe_repo = recipe_repository
        self._subscribers = {}  # user_id -> list of active streams
    
    def StreamUserRecipes(self, request, context):
        """Server Streaming: отправляет рецепты по мере их появления"""
        user_id = request.user_id
        status_filter = request.status_filter
        
        print(f"📡 Клиент {user_id} подписался на поток рецептов")
        
        # Сохраняем подписку
        if user_id not in self._subscribers:
            self._subscribers[user_id] = []
        self._subscribers[user_id].append(context)
        
        try:
            last_recipe_id = None
            while context.is_active():
                # Получаем новые рецепты пользователя
                recipes = self._recipe_repo.find_by_user_id(user_id)
                
                # Фильтруем по статусу
                if status_filter:
                    recipes = [r for r in recipes if r.status.value == status_filter]
                
                # Отправляем новые рецепты
                for recipe in recipes:
                    if recipe.id != last_recipe_id:
                        yield self._to_proto_recipe(recipe)
                        last_recipe_id = recipe.id
                
                time.sleep(2)  # Проверка каждые 2 секунды
                
        finally:
            # Удаляем подписку
            if user_id in self._subscribers:
                self._subscribers[user_id].remove(context)
                if not self._subscribers[user_id]:
                    del self._subscribers[user_id]
            print(f"📡 Клиент {user_id} отписался от потока")
    
    def _to_proto_recipe(self, recipe):
        return recipe_pb2.Recipe(
            id=recipe.id,
            user_id=recipe.user_id,
            name=recipe.name,
            rating=recipe.rating if recipe.rating else None
        )
```

## Контрольные вопросы

**1. В чём преимущество gRPC над REST?**

| № | Преимущество | Описание |
|---|--------------|----------|
| 1 | **Скорость** | Бинарный протокол (Protobuf) быстрее JSON в 3-4 раза |
| 2 | **Размер сообщений** | Данные в Protobuf занимают в 5-10 раз меньше места, чем JSON |
| 3 | **Типобезопасность** | Строгая типизация через .proto схемы, меньше ошибок на этапе выполнения |
| 4 | **Streaming** | Поддержка server-side, client-side и bidirectional streaming из коробки |
| 5 | **HTTP/2** | Мультиплексирование запросов, сжатие заголовков, keep-alive соединения |
| 6 | **Генерация кода** | Автоматическая генерация клиентов и серверов на разных языках |
| 7 | **Контракт** | .proto файл — единый источник правды для API |

**2. Почему Protocol Buffers быстрее JSON?**

| № | Причина | Объяснение |
|---|---------|------------|
| 1 | **Бинарный формат** | Данные передаются в компактном бинарном виде, а не в текстовом |
| 2 | **Отсутствие ключей** | В JSON ключи полей передаются в каждом сообщении, в Protobuf — только номера (1, 2, 3...) |
| 3 | **Фиксированная схема** | Парсинг происходит по известной схеме, без разбора строк и поиска кавычек |
| 4 | **Varint для чисел** | Маленькие числа занимают меньше байт (число 1 → 1 байт, а не 8) |
| 5 | **Типизированные поля** | Длина полей известна заранее, не нужно искать разделители |

**3. Зачем нужен streaming в gRPC?**

| № | Причина | Объяснение |
|---|---------|------------|
| 1 | **Real-time обновления** | Данные доставляются клиенту мгновенно, без необходимости делать повторные запросы |
| 2 | **Экономия памяти** | Не нужно хранить все данные в оперативной памяти перед отправкой |
| 3 | **Эффективная работа с большими данными** | Можно обрабатывать данные по частям, не дожидаясь полной загрузки |
| 4 | **Возможность отмены** | Поток можно прервать в любой момент, если данные больше не нужны |
| 5 | **Двусторонняя связь** | Клиент и сервер могут обмениваться данными одновременно (чат, игры) |
